### 테이블 생성 기본 DDL문 양식

```
USE DABABASE_NM;

-- 테이블 삭제
-- DROP TABLE DABABASE_NM.TABLE_NM;

-- 테이블 생성
CREATE TABLE DABABASE_NM.TABLE_NM
 (
    BOARD_SEQ_NO INT UNSIGNED AUTO_INCREMENT
    ,BOARD_TITLE VARCHAR(255) NOT NULL
    ,BOARD_CTNT LONGTEXT
    ,BOARD_TYPE VARCHAR(2) NOT NULL -- NT : 공지사항, PR : 보도자료, PI : 개인정보
    ,NTC_HIT INT UNSIGNED DEFAULT 0 NOT NULL
    ,EXPS_YN CHAR(1) DEFAULT 'Y' NOT NULL
    ,IMPT_YN CHAR(1) DEFAULT 'N' NOT NULL
    ,ADM_ID VARCHAR(30) NOT NULL
    ,FILE_PATH VARCHAR(255)
    ,MAKE_BSC_DATE DATE NOT NULL
    ,INST_ID VARCHAR(30) NOT NULL
    ,INST_DT TIMESTAMP(3) NOT NULL DEFAULT now(3)
    ,UPDT_ID VARCHAR(30) NOT NULL
    ,UPDT_DT TIMESTAMP(3) NOT NULL DEFAULT now(3)
    ,primary key(BOARD_SEQ_NO)
 )
 ENGINE InnoDB
 CHARACTER SET utf8
 TABLESPACE TSDE001
 ;

-- 인덱스 생성
-- ALTER TABLE DABABASE_NM.TABLE_NM DROP INDEX UX01_HOMEPAGE_BOARD;
-- ALTER TABLE DABABASE_NM.TABLE_NM DROP INDEX IX01_HOMEPAGE_BOARD;

CREATE UNIQUE INDEX UX01_HOMEPAGE_BOARD ON DABABASE_NM.TABLE_NM(BOARD_SEQ_NO);
CREATE INDEX IX01_HOMEPAGE_BOARD ON DABABASE_NM.TABLE_NM(BOARD_TYPE, EXPS_YN, BOARD_SEQ_NO);

-- 코멘트 지정
ALTER TABLE DABABASE_NM.TABLE_NM COMMENT = '예시';

ALTER TABLE DABABASE_NM.TABLE_NM MODIFY BOARD_SEQ_NO INT UNSIGNED AUTO_INCREMENT COMMENT '게시판일련번호';
ALTER TABLE DABABASE_NM.TABLE_NM MODIFY BOARD_TITLE VARCHAR(255) NOT NULL COMMENT '게시판제목';
ALTER TABLE DABABASE_NM.TABLE_NM MODIFY BOARD_CTNT LONGTEXT COMMENT '게시판내용';
ALTER TABLE DABABASE_NM.TABLE_NM MODIFY BOARD_TYPE VARCHAR(2) NOT NULL COMMENT '게시판유형 -- NT : 공지사항, PR : 보도자료, PI : 개인정보';
ALTER TABLE DABABASE_NM.TABLE_NM MODIFY NTC_HIT INT UNSIGNED DEFAULT 0 NOT NULL COMMENT '조회수';
ALTER TABLE DABABASE_NM.TABLE_NM MODIFY EXPS_YN CHAR(1) DEFAULT 'Y' NOT NULL COMMENT '노출여부';
ALTER TABLE DABABASE_NM.TABLE_NM MODIFY IMPT_YN CHAR(1) DEFAULT 'N' NOT NULL COMMENT '중요여부';
ALTER TABLE DABABASE_NM.TABLE_NM MODIFY ADM_ID VARCHAR(30) NOT NULL COMMENT '운영자ID';
ALTER TABLE DABABASE_NM.TABLE_NM MODIFY FILE_PATH VARCHAR(255) COMMENT '파일경로';
ALTER TABLE DABABASE_NM.TABLE_NM MODIFY MAKE_BSC_DATE DATE NOT NULL COMMENT '작성기준일';
ALTER TABLE DABABASE_NM.TABLE_NM MODIFY INST_ID VARCHAR(30) NOT NULL COMMENT '입력ID';
ALTER TABLE DABABASE_NM.TABLE_NM MODIFY INST_DT TIMESTAMP(3) NOT NULL DEFAULT now(3) COMMENT '입력일시';
ALTER TABLE DABABASE_NM.TABLE_NM MODIFY UPDT_ID VARCHAR(30) COMMENT '수정ID';
ALTER TABLE DABABASE_NM.TABLE_NM MODIFY UPDT_DT TIMESTAMP(3) NOT NULL DEFAULT now(3) COMMENT '수정일시';
```

* MySQL의 경우 TIMESTAMP 값이 second 가 default인 관계로 자리수 지정하여 millisecond 까지 지정 필요
* oracle과 달리 comment 지정 시에 column modify 방식 사용하며 컬럼 속성도 재지정되므로 동일하게 작성 필요
